      - mydb
    restart: always
    networks:
      - back-net
      - front-net
  myweb:
    build:
      context: ./frontend
      dockerfile: Dockerfile
    environment:
      API_IP: myapp:8080
    ports:
      - "3000:3000"
    depends_on:
      - myapp
